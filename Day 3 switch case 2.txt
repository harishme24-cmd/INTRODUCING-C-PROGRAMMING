#include <stdio.h>
void main()
{
    int a,b,c,d,e;
    int choice,sum,max;
    printf("Enter 5 number:\n");
    scanf("%d %d %d %d %d",&a,&b,&c,&d,&e);
    printf("\n choose on option:\n");
    printf("1.Display all numbers\n");
    printf("2.Find sum of numbers\n");
    printf("3.Find maximum numbers\n");
    printf("4.Ex:\n");
    printf("Enter your choice:");
    scanf("%d",&choice);
    switch(choice)
    {
        case 1:
        printf("\n Numbers are:%d %d %d %d %d\n",a,b,c,d,e);
        break;
        case 2:
        sum=a+b+c+d+e;
        printf("\nsum =%d\n",sum);
        break;
        case 3:
        max=a;
        if(b>max)max=b;
        if(c>max)max=c;
        if(d>max)max=d;
        if(e>max)max=e;
        printf("\n Maximum=%d \n",max);
        break;
        default:
        printf("\nInvalid choice:\n");
    }
}
