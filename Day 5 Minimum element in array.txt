#include <stdio.h>

int main() {
    int arr[] = {12, 5, 7, 3, 19, -4, 8};
    int n = sizeof(arr) / sizeof(arr[0]);
    int min = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] < min)
            min = arr[i];
    }
    printf("The minimum element in the array is %d\n", min);
    return 0;
}
